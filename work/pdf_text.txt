/**
 * 3. Dynamic Kinetic Burn Simulator
 * Calculates the exact minutes of physical activities required to burn a specific calorie 
delta (surplus)
 */
export function calculateCardioOffsets(caloricDelta: number, weightKg: number): CardioTargets {
  if (caloricDelta <= 0) {
    return { runningMins: 0, cyclingMins: 0, walkingMins: 0 };
  }
  // MET Constants
  const MET_RUNNING = 8.80000000;
  const MET_CYCLING = 6.80000000;
  const MET_WALKING = 4.30000000;
  // Formula: Burn Rate (kcal/min) = (MET * 3.5 * weightKg) / 200
  const runningBurnRate = (MET_RUNNING * 3.5 * weightKg) / 200;
  const cyclingBurnRate = (MET_CYCLING * 3.5 * weightKg) / 200;
  const walkingBurnRate = (MET_WALKING * 3.5 * weightKg) / 200;
  // Duration = Delta / Burn Rate
